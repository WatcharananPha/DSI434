python
import re
import json

def to_snake_case(name):
    """Converts a string to snake_case."""
    # Replace non-alphanumeric/space with underscore, then strip leading/trailing underscores
    name = re.sub(r'[^a-zA-Z0-9\s]+', '_', name)
    name = re.sub(r'\s+', '_', name).strip('_')
    # Convert CamelCase to snake_case (e.g., 'FooBar' -> 'foo_bar')
    name = re.sub(r'(?<!^)(?=[A-Z])', '_', name).lower()
    return name.lower()

def clean_header_part(part):
    """
    Cleans a raw header part by stripping quotes, collapsing newlines to spaces,
    trimming whitespace, and prioritizing English text if both English and Thai are present
    or if there's an explanation in parentheses.
    """
    part = part.strip()

    # Strip quotes if present
    if part.startswith('"') and part.endswith('"'):
        part = part[1:-1]
    
    # Collapse internal newlines to a single space
    part = part.replace('\n', ' ').strip()

    # Handle "(explanation)" pattern for brevity in key names
    # e.g., "Mitigation Plan (new countermeasure...)" -> "Mitigation Plan"
    parentheses_match = re.match(r'^(.*?)\s*\(.*\)$', part)
    if parentheses_match:
        part = parentheses_match.group(1).strip()
    
    # Prioritize English text if the string contains both English and Thai,
    # assuming English comes first.
    # This regex captures an initial segment of English letters, numbers, spaces,
    # and common punctuation, followed by optional non-English characters.
    contains_thai = any(0x0E00 <= ord(c) <= 0x0E7F for c in part)
    contains_english = bool(re.search(r'[a-zA-Z]', part))

    if contains_thai and contains_english:
        english_part_match = re.match(r'^([A-Za-z0-9\s&()_/-]+)\s*[\u0E00-\u0E7F]*.*$', part, re.UNICODE)
        if english_part_match:
            part = english_part_match.group(1).strip()

    return part

def process_tsv_to_jsonl(tsv_data, sheet_name):
    lines = tsv_data.strip().split('\n')
    
    if len(lines) < 2:
        raise ValueError("TSV data must contain at least two header lines.")

    # Read the two header lines
    raw_h1_parts = lines[0].split('\t')
    raw_h2_parts = lines[1].split('\t')
    
    header_keys = []
    key_counts = {}
    
    # Determine the maximum number of columns across both header lines
    num_cols = max(len(raw_h1_parts), len(raw_h2_parts))

    for i in range(num_cols):
        part1 = raw_h1_parts[i] if i < len(raw_h1_parts) else ''
        part2 = raw_h2_parts[i] if i < len(raw_h2_parts) else ''

        cleaned_part1 = clean_header_part(part1)
        cleaned_part2 = clean_header_part(part2)
        
        combined_key_candidate = ''
        if cleaned_part1:
            combined_key_candidate = cleaned_part1
        
        if cleaned_part2:
            if combined_key_candidate:
                # Combine part1 and part2 if both are present
                combined_key_candidate = f"{combined_key_candidate} {cleaned_part2}"
            else:
                # If only part2 is present, use it
                combined_key_candidate = cleaned_part2

        # If after combining, the key candidate is still empty, use a generic column name
        if not combined_key_candidate:
            final_key = f"column_{i+1}"
        else:
            final_key = to_snake_case(combined_key_candidate)
            
            # Ensure uniqueness for generated keys
            base_key = final_key
            key_counts[base_key] = key_counts.get(base_key, 0) + 1
            if key_counts[base_key] > 1:
                final_key = f"{base_key}_{key_counts[base_key]}"
        
        header_keys.append(final_key)

    json_lines_output = []
    row_index_counter = 0 # 1-based index for valid data rows
    
    # Process data rows starting from the third line (index 2)
    for line in lines[2:]:
        raw_values = line.split('\t')
        
        data_object = {}
        all_empty_columns = True

        for i in range(min(len(header_keys), len(raw_values))):
            key = header_keys[i]
            value = raw_values[i]
            
            # Trim leading/trailing spaces
            value = value.strip()
            
            # Collapse internal newlines to a single space
            value = value.replace('\n', ' ')
            
            # Add to data_object only if value is not empty after processing
            if value:
                data_object[key] = value
                all_empty_columns = False
        
        # Drop rows with all-empty columns
        if not all_empty_columns:
            row_index_counter += 1
            json_object = {
                "sheet": sheet_name,
                "row_index": row_index_counter,
                "data": data_object
            }
            # Use ensure_ascii=False to correctly handle Thai characters
            json_lines_output.append(json.dumps(json_object, ensure_ascii=False))
            
    return "\n".join(json_lines_output)
