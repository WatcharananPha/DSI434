python
import json
import re

def convert_tsv_to_json_lines(tsv_string: str, sheet_name: str = "2._BIA_Process") -> str:
    """
    Converts a TSV string to JSON Lines format according to specified rules.

    Args:
        tsv_string: The input TSV data as a single string.
        sheet_name: The fixed sheet name to be included in each JSON object.

    Returns:
        A string containing JSON objects, one per line.
    """
    json_lines_output = []
    
    # Split the TSV string into individual lines, trimming overall leading/trailing whitespace
    lines = tsv_string.strip().split('\n')

    if not lines:
        return ""

    # --- 1. Process header row ---
    header_line = lines[0]
    raw_headers = header_line.split('\t')

    def to_snake_case(header_text: str) -> str:
        """Converts a header string to snake_case, handling spaces, newlines, and special chars."""
        # Trim and remove surrounding quotes
        s = header_text.strip().replace('"', '')
        # Collapse all whitespace (including newlines) to a single space
        s = re.sub(r'\s+', ' ', s).strip()
        # Convert to lowercase
        s = s.lower()
        # Replace anything that's NOT a lowercase letter, digit, or Thai character with '_'
        # \u0e00-\u0e7f covers most Thai characters
        s = re.sub(r'[^a-z0-9\u0e00-\u0e7f]+', '_', s)
        # Remove leading/trailing underscores that might result from previous steps
        s = s.strip('_')
        # Collapse multiple underscores into a single one
        s = re.sub(r'_+', '_', s)
        return s

    headers = [to_snake_case(h) for h in raw_headers]

    # --- 2. Process data rows ---
    row_index_counter = 0
    for line in lines[1:]: # Start from the second line (data rows)
        raw_cells = line.split('\t')
        processed_cells = []
        
        # Flag to check if the entire row is empty after processing cells
        # This is for the "drop rows with all-empty columns" requirement.
        all_cells_empty_in_row = True 

        for cell in raw_cells:
            # Check if the cell content is quoted, and remove outer quotes if so
            is_quoted = False
            if cell.startswith('"') and cell.endswith('"') and len(cell) >= 2:
                cell_content = cell[1:-1]
                is_quoted = True
            else:
                cell_content = cell
            
            # Collapse all internal whitespace (including newlines) to a single space
            # Then trim leading/trailing spaces from the result
            processed_cell_value = re.sub(r'\s+', ' ', cell_content).strip()
            
            processed_cells.append(processed_cell_value)
            
            # If any cell has content, the row is not considered "all-empty"
            if processed_cell_value:
                all_cells_empty_in_row = False
        
        # Drop rows where all columns are empty after processing
        if all_cells_empty_in_row:
            continue
        
        row_index_counter += 1 # Increment 1-based row index for valid data rows
        
        data_object = {}
        for i, header in enumerate(headers):
            if i >= len(processed_cells): # Handle cases where a row might have fewer columns than headers
                cell_value = "" # Treat missing cells as empty
            else:
                cell_value = processed_cells[i]

            # Attempt type conversion: int -> float -> string
            if cell_value == "":
                data_object[header] = "" # Keep empty string as is
            else:
                try:
                    data_object[header] = int(cell_value)
                except ValueError:
                    try:
                        data_object[header] = float(cell_value)
                    except ValueError:
                        data_object[header] = cell_value # Preserve as string if not a number

        # Construct the final JSON object for the line
        json_object = {
            "sheet": sheet_name,
            "row_index": row_index_counter,
            "data": data_object
        }
        
        # Append the JSON object as a string to the list.
        # ensure_ascii=False ensures Thai characters are preserved directly.
        json_lines_output.append(json.dumps(json_object, ensure_ascii=False))

    return "\n".join(json_lines_output)
